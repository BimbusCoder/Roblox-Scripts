
local library = loadstring(game:HttpGet("https://pastebin.com/raw/tqKTJQAA", true))()

local Window = library:CreateWindow('A Modded Adventure')

Window:Section('Item Farm')


local Toggle = Window:Toggle("Hyper Gem", {flag = "ThisNeedsToBeRemembered1"})
spawn( 
    function() --Makes this a new loop within the running script so it doesn't interfere with the rest of the script
        while wait(1) do
            if Window.flags.ThisNeedsToBeRemembered1 then
                for _,v in pairs(game.Workspace:GetChildren()) do
                    if v:IsA("Tool") and v.Name == "Hyper Gem" then
                        v.Handle.CFrame = game.Players.LocalPlayer.Character.PrimaryPart.CFrame
                        wait()
                    end
                end
            end
        end
    end
)
